SELECT 
  estacion.codigo, 
  estacion.tipo, 
  estacion.clase, 
  estacion.cat, 
  estacion.nombre, 
  estacion.municipio, 
  estacion.corriente, 
  estacion.departamento, 
  estacion.altitud, 
  estacion.cod_dep, 
  estacion.cod_muni, 
  estacion.longitud, 
  estacion.latitud, 
  estacion.estado, 
  estacion.geom,
  tmp_query_mensual.variable,
  avg(tmp_query_mensual.enero) enero, 
  avg(tmp_query_mensual.febrero) febrero, 
  avg(tmp_query_mensual.marzo) marzo, 
  avg(tmp_query_mensual.abril) abril, 
  avg(tmp_query_mensual.mayo) mayo, 
  avg(tmp_query_mensual.junio) junio, 
  avg(tmp_query_mensual.julio) julio, 
  avg(tmp_query_mensual.agosto) agosto, 
  avg(tmp_query_mensual.septiembre) septiembre, 
  avg(tmp_query_mensual.diciembre) diciembre, 
  avg(tmp_query_mensual.noviembre) noviembre, 
  avg(tmp_query_mensual.octubre) octubre, 
  tmp_query_mensual.cod_estacion
  
FROM 
  public.tmp_query_mensual, 
  public.estacion
WHERE 
  estacion.codigo = tmp_query_mensual.cod_estacion AND   tmp_query_mensual.variable = '&&&' 

GROUP BY 
  estacion.codigo,
   tmp_query_mensual.cod_estacion,    
  estacion.tipo, 
  estacion.clase, 
  estacion.cat, 
  estacion.nombre, 
  estacion.municipio, 
  estacion.corriente, 
  estacion.departamento, 
  estacion.altitud, 
  estacion.cod_dep, 
  estacion.cod_muni,
  tmp_query_mensual.variable
